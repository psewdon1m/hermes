Development HTTP file server:
node -e "const { createServer } = require('http'); const { createReadStream, statSync } = require('fs'); const { extname, join } = require('path'); const types = { '.html': 'text/html; charset=utf-8', '.js': 'text/javascript; charset=utf-8', '.css': 'text/css; charset=utf-8', '.json': 'application/json; charset=utf-8', '.png': 'image/png', '.jpg': 'image/jpeg', '.svg': 'image/svg+xml' }; createServer((req, res) => { const p = req.url === '/' ? '/index.html' : req.url; const file = join(process.cwd(), decodeURIComponent(p)); try { const stat = statSync(file); if (stat.isDirectory()) throw new Error('dir'); const ext = extname(file); res.writeHead(200, { 'Content-Type': types[ext] || 'application/octet-stream' }); createReadStream(file).pipe(res); } catch (e) { res.writeHead(404); res.end('Not found'); } }).listen(3000, () => console.log('http://localhost:3000'));"
